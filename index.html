// ══ PUSH TO HOSTAWAY ══════════════════════════════════════════════════════════════


async function pushToHostaway(){


  let hwToken;


  try{ hwToken = await getHwToken(); }


  catch(e){ notify('Hostaway token hiba: '+e.message,'err'); nav('conn'); return; }


  const l = S.listings.find(x=>x.id===S.selId);


  if(!l){ notify('Nincs kiválasztott lakás','err'); return; }


  const p = S.prices[S.selId];


  if(!p||Object.keys(p).filter(k=>!k.startsWith('_')).length===0){


    notify('Nincs ár – futtass piac kutatást vagy add meg az alap árat','err'); return;


  }


 


  // collect future dates only


  const today = new Date().toISOString().split('T')[0];


  const entries = Object.entries(p)


    .filter(([k,v])=>!k.startsWith('_') && k>=today && v>0)


    .sort(([a],[b])=>a.localeCompare(b));


 


  if(!entries.length){ notify('Nincs jövőbeli ár a feltöltéshez','err'); return; }


 


  // confirm


  const confirmed = confirm(


    `${l.name}\n\n${entries.length} nap árait töltöd fel Hostawaybe (${entries[0][0]} – ${entries[entries.length-1][0]}).\n\nFolytatod?`


  );


  if(!confirmed) return;


 


  const btns = [document.getElementById('btn-push'), document.getElementById('cal-push-btn')];


  btns.forEach(b=>{ if(b){ b.disabled=true; b.innerHTML='<span class="spin"></span>Feltöltés...'; } });


  document.getElementById('bp-status').textContent='Hostaway feltöltés folyamatban...';


 


  // Hostaway API: PATCH /v1/listings/{id}/calendar in batches of 30


  const BATCH = 30;


  let ok=0, fail=0;


 


  for(let i=0; i<entries.length; i+=BATCH){


    const batch = entries.slice(i, i+BATCH);


    const calendarData = batch.map(([date, price])=>({


      date,


      price: Math.round(price),


      isAvailable: 1,


      minimumStay: 1,


    }));


 


    try{


      const hwPushBase=location.hostname==='localhost'?'/api/hostaway':'https://api.hostaway.com';


      const res = await fetch(`${hwPushBase}/v1/listings/${l.id}/calendar`, {


        method: 'PUT',


        headers: {


          'Authorization': `Bearer ${hwToken}`,


          'Content-Type': 'application/json',


          'Cache-control': 'no-cache',


        },


        body: JSON.stringify({ days: calendarData })


      });


      const d = await res.json();


      if(res.ok || d.status==='success'){ ok+=batch.length; }


      else{


        // try alternative endpoint format


        const res2 = await fetch(`${hwPushBase}/v1/listings/${l.id}/calendar`, {


          method: 'PATCH',


          headers: { 'Authorization': `Bearer ${hwToken}`, 'Content-Type': 'application/json' },


          body: JSON.stringify(calendarData)


        });


        if(res2.ok) ok+=batch.length; else fail+=batch.length;


      }


    }catch(e){ fail+=batch.length; }


 


    // progress


    document.getElementById('bp-status').textContent=


      `Feltöltés: ${Math.min(i+BATCH,entries.length)}/${entries.length} nap... (${ok} sikeres, ${fail} hiba)`;


 


    // small delay between batches


    if(i+BATCH<entries.length) await new Promise(r=>setTimeout(r,300));


  }


 


  btns.forEach(b=>{ if(b){ b.disabled=false; b.innerHTML=b.id==='btn-push'?'↑ Feltöltés Hostawaybe':'↑ Push Hostawaybe'; } });


 


  if(fail===0){


    document.getElementById('bp-status').textContent=`✓ ${ok} nap sikeresen feltöltve Hostawaybe – ${new Date().toLocaleString('hu-HU')}`;


    notify(`${ok} nap ár feltöltve Hostawaybe ✓`);


  } else if(ok>0){


    document.getElementById('bp-status').textContent=`⚠ ${ok} sikeres, ${fail} sikertelen. Ellenőrizd a Hostaway API jogosultságokat.`;


    notify(`${ok} feltöltve, ${fail} hiba – ellenőrizd a Hostaway jogokat`,'err');


  } else {


    document.getElementById('bp-status').textContent='✗ Feltöltés sikertelen – ellenőrizd a Hostaway API key-t és a listing ID-t';


    notify('Hostaway feltöltés sikertelen – lehet hogy csak olvasási jog van a key-hez','err');


  }


}
